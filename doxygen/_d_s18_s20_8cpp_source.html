99 uint16_t DS18S20::convertToRaw(uint8_t* data, uint8_t size, int8_t type)
100 {
101  unsigned int raw = (data[1] << 8) | data[0];
102  if (type == 1)
103  {
104  raw = raw << 3; // 9 bit resolution default
105  if (data[7] == 0x10)
106  {
107  // count remain gives full 12 bit resolution
108  raw = (raw & 0xFFF0) + 12 - data[6];
109  }
110  }
111  else
112  {
113  byte cfg = (data[4] & 0x60);
114  if (cfg == 0x00)
115  raw = raw << 3; // 9 bit resolution, 93.75 ms
116  else if (cfg == 0x20)
117  raw = raw << 2; // 10 bit res, 187.5 ms
118  else if (cfg == 0x40)
119  raw = raw << 1; // 11 bit res, 375 ms
120  // default is 12 bit resolution, 750 ms conversion time
121  }
122  return raw;
123 }
