196 void DigitalOutput::setPWM(float dutyCycle)
197 {
198  if (supportsPWM() && dutyCycle <= 1.0)
199  {
200  m_DutyCycle = dutyCycle;
201  if (m_OnValue == 0)
202  analogWrite(m_Pin, (uint8_t) ((1.0 - dutyCycle) * 255));
203  else
204  analogWrite(m_Pin, (uint8_t) (dutyCycle * 255));
205  if (m_DutyCycle > 0)
206  m_On = 1;
207  else
208  m_On = 0;
209  }

222 int8_t DigitalOutput::showWebinterface(WebServer* server,
223  WebServer::ConnectionType type,
224  char* url)
225 {
226  File templateFile;
227  TemplateParser* parser;
228  int16_t matchIdx;
229  char templateFileName[template_digitaloutput_fnsize];
230  strcpy_P(templateFileName, template_digitaloutput_fname);
231  float dC = -1.0;
232 
233  if (type == WebServer::POST)
234  {
235  int8_t repeat;
236  char name[16], value[16];
237  do
238  {
239  repeat = server->readPOSTparam(name, 16, value, 16);
240  if (strcmp_P(name,
241  (PGM_P) pgm_read_word(&(template_digitaloutput_inputs[DO_I_ON])))
242  == 0)
243  {
244  m_OnValue = atoi(value);
245  m_OffValue = 1 - m_OnValue;
246  }
247  else if (strcmp_P(name,
248  (PGM_P) pgm_read_word(&(template_digitaloutput_inputs[DO_I_PIN])))
249  == 0)
250  {
251  m_Pin = atoi(value);
252  pinMode(m_Pin, OUTPUT);
253  }
254  else if (strcmp_P(name,
255  (PGM_P) pgm_read_word(&(template_digitaloutput_inputs[DO_I_PWM])))
256  == 0)
257  {
258  dC = atof(value);
259  if (dC < 0.0)
260  dC = 0;
261  else if (dC > 100.0)
262  dC = 100.0;
263  dC /= 100;
264  }
265 
266  } while (repeat);
267  if (dC >= 0)
268  setPWM(dC);
269  server->httpSeeOther(this->m_URL);
270  }
271  else
272  {
273  server->httpSuccess();
274  parser = aquaduino->getTemplateParser();
275  templateFile = SD.open(templateFileName, FILE_READ);
276  while ((matchIdx =
277  parser->processTemplateUntilNextMatch(&templateFile,
278  template_digitaloutput,
279  template_digitaloutput_elements,
280  server))
281  >= 0)
282  {
283  switch (matchIdx)
284  {
285  case DO_ANAME:
286  server->print(getName());
287  break;
288  case DO_PIN_NAME:
289  server->print((const __FlashStringHelper *) (pgm_input_pin));
290  break;
291  case DO_PIN_VAL:
292  server->print(m_Pin);
293  break;
294  case DO_PIN_SIZE:
295  server->print(3);
296  break;
297  case DO_PIN_MAXLENGTH:
298  server->print(2);
299  break;
300  case DO_PWM_NAME:
301  server->print((const __FlashStringHelper *) (pgm_input_pwm));
302  break;
303  case DO_PWM_VAL:
304  if (supportsPWM())
305  {
306  server->print(getPWM() * 100);
307  }
308  else
309  {
310  server->print(F("No PWM"));
311  }
312  break;
313  case DO_PWM_SIZE:
314  server->print(7);
315  break;
316  case DO_PWM_MAXLENGTH:
317  server->print(6);
318  break;
319  case DO_ON_NAME:
320  server->print((const __FlashStringHelper *) (pgm_input_on));
321  break;
322  case DO_ON_OPTIONS:
323  parser->selectListOption("LOW", "0", m_OnValue == 0, server);
324  parser->selectListOption("HIGH", "1", m_OnValue == 1, server);
325  break;
326  }
327  }
328 
329  templateFile.close();
330  }
331  return true;
