100 void printActuatorTable(WebServer* server)
101 {
102  int i, j;
103  int16_t matchIdx = 0;
104  TemplateParser* parser;
105  char actuatorID[5], controllerID[5], lockedID[5], stateID[5];
106  char aRowFileName[sizeof(pARowFileName)];
107  File templateARow;
108  Actuator* currentActuator;
109  Controller* currentController;
110 
111  parser = aquaduino->getTemplateParser();
112  strcpy_P(aRowFileName, pARowFileName);
113 
114  aquaduino->resetActuatorIterator();
115 
116  while ((i = aquaduino->getNextActuator(&currentActuator)) != -1)
117  {
118  templateARow = SD.open(aRowFileName, FILE_READ);
119  while ((matchIdx =
120  parser->processTemplateUntilNextMatch(&templateARow,
121  actuatorTemplate,
122  sizeof(actuatorTemplate) / sizeof(char*),
123  server))
124  != -1)
125  {
126 
127  switch (matchIdx)
128  {
129  case A_COLOR:
130  if (i % 2 == 0)
131  {
132  server->print("#FFFFFF");
133  }
134  else
135  {
136  server->print("#99CCFF");
137  }
138  break;
139  case A_IACTUATOR:
140  actuatorID[0] = 'A';
141  itoa(i, &actuatorID[1], 10);
142  server->print(actuatorID);
143  break;
144  case A_ACTUATORNAME:
145  server->print(currentActuator->getName());
146  break;
147  case A_CSELECT:
148  controllerID[0] = 'C';
149  itoa(i, &controllerID[1], 10);
150  server->print(controllerID);
151  break;
152  case A_COPTIONS:
153  aquaduino->resetControllerIterator();
154  while ((j = aquaduino->getNextController(&currentController)) != -1)
155  {
156  itoa(j, controllerID, 10);
157  parser->selectListOption(currentController->getName(),
158  controllerID,
159  currentActuator->getController() == j,
160  server);
161  }
162  break;
163  case A_LSELECT:
164  lockedID[0] = 'L';
165  itoa(i, &lockedID[1], 10);
166  server->print(lockedID);
167  break;
168  case A_LOPTIONS:
169  parser->selectListOption("Unlocked", "0", 0, server);
170  parser->selectListOption("Locked",
171  "1",
172  currentActuator->isLocked(),
173  server);
174  break;
175  case A_SSELECT:
176  stateID[0] = 'S';
177  itoa(i, &stateID[1], 10);
178  server->print(stateID);
179  break;
180  case A_SOPTIONS:
181  parser->selectListOption("Off", "0", 0, server);
182  parser->selectListOption("On",
183  "1",
184  currentActuator->isOn(),
185  server);
186  break;
187  case A_LINK:
188  actuatorID[0] = 'A';
189  itoa(i, &actuatorID[1], 10);
190  server->print(actuatorID);
191  break;
192  }
193  }
194  templateARow.close();
195  }

197 }

241 void defaultCmd(WebServer &server, WebServer::ConnectionType type, char *, bool)
242 {
243  TemplateParser* parser;
244  char fileName[sizeof(pMFileName)];
245  File templateFile;
246  int16_t matchIdx;
247  char name[30], value[30];
248  uint16_t actuatorIdx;
249  uint16_t controllerIdx;
250  Actuator* actuator;
251  int8_t repeat = 0;
252 
253  parser = aquaduino->getTemplateParser();
254 
255  strcpy_P(fileName, pMFileName);
256 
257  templateFile = SD.open(fileName, FILE_READ);
258 
259  if (type == WebServer::POST)
260  {
261  while ((repeat = server.readPOSTparam(name, 30, value, 30)) > 0)
262  {
263 
264  if (name[0] == 'A' && name[1] >= '0' && name[1] <= '9')
265  {
266  actuatorIdx = atoi(&name[1]);
267  aquaduino->getActuator(actuatorIdx)->setName(value);
268  }
269  else if (name[0] == 'C' && name[1] >= '0' && name[1] <= '9')
270  {
271  actuatorIdx = atoi(&name[1]);
272  controllerIdx = atoi(value);
273  aquaduino->getActuator(actuatorIdx)->setController(controllerIdx);
274  }
275  else if (name[0] == 'L' && name[1] >= '0' && name[1] <= '9')
276  {
277  actuatorIdx = atoi(&name[1]);
278  if (atoi(value) == 1)
279  aquaduino->getActuator(actuatorIdx)->lock();
280  else
281  aquaduino->getActuator(actuatorIdx)->unlock();
282  }
283  else if (name[0] == 'S' && name[1] >= '0' && name[1] <= '9')
284  {
285  actuatorIdx = atoi(&name[1]);
286  if ((atoi(value) == 1) && (aquaduino->getActuator(actuatorIdx)->getController()
287  == -1))
288  aquaduino->getActuator(actuatorIdx)->on();
289  else
290  aquaduino->getActuator(actuatorIdx)->off();
291  }
292  }
293 
294  aquaduino->resetActuatorIterator();
295  while (aquaduino->getNextActuator(&actuator) != -1)
296  {
297  aquaduino->writeConfig(actuator);
298  }
299 
300  server.httpSeeOther("/");
301 
302  }
303  else if (type != WebServer::HEAD)
304  {
305  server.httpSuccess();
306 
307  matchIdx = 0;
308 
309  while ((matchIdx =
310  parser->processTemplateUntilNextMatch(&templateFile,
311  mainTemplate,
312  sizeof(mainTemplate) / sizeof(char*),
313  &server))
314  >= 0)
315  {
316  switch (matchIdx)
317  {
318  case M_TEMPERATURE:
319  server.print(aquaduino->getTemperature());
320  break;
321  case M_HOUR:
322  server.print(hour());
323  break;
324  case M_MINUTE:
325  server.print(minute());
326  break;
327  case M_SECOND:
328  server.print(second());
329  break;
330  case M_CONTROLLER:
331  printControllerTable(&server);
332  break;
333  case M_ACTUATOR:
334  printActuatorTable(&server);
335  break;
336  }
337  }
338  }
339  templateFile.close();
340 }
