778 int8_t Aquaduino::getAssignedActuators(Controller* controller,
779  Actuator** actuators, int8_t max)
780 {
781  int8_t actuatorIdx = -1;
782  int8_t nrOfAssignedActuators = 0;
783  Actuator* currentActuator;
784  int8_t controllerIdx = m_Controllers.findElement(controller);
785 
786  m_Actuators.resetIterator();
787  while ((actuatorIdx = m_Actuators.getNext(&currentActuator)) != -1)
788  {
789  if (currentActuator->getController() == controllerIdx)
790  {
791  if (nrOfAssignedActuators < max)
792  actuators[nrOfAssignedActuators] = currentActuator;
793  nrOfAssignedActuators++;
794  }
795  }
796  return nrOfAssignedActuators;
797 }
